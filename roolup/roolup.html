roolup #

  • 基本配置
export default{
  input:'src/index.js',
  plugins:[
  ],
  output:{
    file:'cjs.bundle.js',
    /**
     * iife  立即执行包  ，适用于作为<script>标签，如果要为应用程序创建一个捆绑包，您可能想要使用它，因为它会使文件大小变小
     * amd  异步模块定义，用于像RequireJS 这样的模块加载器
     * esm  打包为es module模块文件- 在现代浏览器中可以通过 <script type=module> 标签引入
     * cjs  commonJS 适用于nodejs
     * umd  通用模块定义，以amd cjs  iife为一体
     * system SystemJS加载器格式  //生成的文件中，第三个参数返回的是一个execute执行器
     */
    format:'cjs',
    name:'myBundle'
  }
}

  • 插件配置
import rollupPluginMyexample from "./plugins/rollup-plugin-my-example.js";
import rollupNodeResolve from "@rollup/plugin-node-resolve";
import rollupCommonJS from "@rollup/plugin-commonjs";
import rollupTestPlugin from "./plugins/@rollup/plugin-test";

export default {
  input: "src/index.js",
  plugins: [
    /**
     * 插件是从上往下执行
     * rollupCommonJS要放到最上面，应该用在其他插件转换你的模块之前 - 这是为了防止其他插件的改变破坏源文件，导致CommonJS检测失效，这个插件就没啥用了。
     */
    rollupCommonJS(),
    rollupNodeResolve({
      // 将自定义选项传递给解析插件
      moduleDirectories: ['node_modules'],
    }),
    rollupTestPlugin(),
    rollupPluginMyexample(),
  ],
  output: {
    file: "bundlePlugin.js",
    /**
     * iife  立即执行包  ，适用于作为<script>标签，如果要为应用程序创建一个捆绑包，您可能想要使用它，因为它会使文件大小变小
     * amd  异步模块定义，用于像RequireJS 这样的模块加载器
     * esm  打包为es module模块文件- 在现代浏览器中可以通过 <script type=module> 标签引入
     * cjs  commonJS 适用于nodejs
     * umd  通用模块定义，以amd cjs  iife为一体
     * system SystemJS加载器格式  //生成的文件中，第三个参数返回的是一个execute执行器
     */
    format: "cjs",
    name: "myBundle",
  },
};
  • 自定义插件
export default function () {
  return {
    name: "@rollup/plugin-test",
    options(...args) {
      console.log("我自己写的插件,构建的第一个钩子 ", args);
    },
    buildStart() {
      console.log("我自己写的插件,开始构建");
    },
    resolveId(source) {
      console.log("这是我自己写的插件", source);
      return source;
    },
    load(id) {
      if (id.includes("bar")) {
        console.log("我自己写的插件，load钩子 ", id);
        return "export default ''";
      }
      return null;
    },
  };
}
