一、基本数据类型中 null 和 undefined 区别及其使用 #

1.1、都是代表没值 #

  • null 表示没有对象，该处不应该有值
  • undefined 表示缺少值

1.2、转为数值不同 #

  • null 转为数值为 0

  • undefined 转为数值是 NaN

    Number(null);
0;
Number(undefined);
NaN;

1.3、应用场景 #

  • 什么时候为 null

    • 作为函数参数，表示该函数的参数不是对象
    • 作为对象原型链的终点
    Object.getPrototypeOf(Object.prototype);
null;

  • 什么时候出现 undefined

    • 变量声明了，但是没有赋值，就等于 undefined
    var a;

console.log(a);

  • 调用函数时，应该提供的参数没有提供，该参数就等于 undefined

    function fn(a) {
  console.log(a);
}

fn();

  • 对象没有赋值的属性，该属性的值为 undefined

    var ob = {}

console.log(ob.name)

  • 函数没有返回值，默认返回 undefined

    var f = fn(1)

console.log(f)

二、js 中检测数据类型的方式 #

2.1、typeof #

  • 检测数据类型的运算符

  • 数组、对象 、null 都是 objece

typeof null === 'object'

在 JavaScript 最初的实现中，JavaScript 中的值是由一个表示类型的标签和实际数据值表示的。对象的类型标签是 0。由于 null 代表的是空指针（大多数平台下值为 0x00），因此，null 的类型标签是 0，typeof null 也因此返回 "object"。

2.2、instanceof #

  • 检测某一个实例是否属于这个类

  • 可以正确判断对象的类型，内部运行机制是判断在他的原型链上是否能找到这个类型的原型

  • 判断构造函数的 prototype 属性是否出现在对象的原型链中的任何位置

10 instanceof Number --> false
true instanceof Boolean  ---->false
'hello' instanceof String ---->false

[] instanceof Array---->true
function(){} instanceof Function--->true
{} instanceof Object ---> true


function Person(name){
    this.name = name
}

let pp = new Person('zs')

Object.getPrototypeOf(pp) === Person.prototype
Object.getPrototypeOf(Object.getPrototypeOf(pp)) === Object.protptype
  • 实现 instanceof
function instance(left, right) {
  //获取对象原型
  let proto = Object.getPrototypeOf(left);
  //获取构造函数的prototype属性
  let prototype = right.prototype;
  while (true) {
    if (!proto) {
      //像基本数据类型就直接返回false
      return false;
    }
    if (proto === prototype) {
      return true;
    }
    //为啥递归？
    //解决console.log(pp instance Object)
    //如果没找到，就继续在原型链上去找，只要找到位置就是true
    proto = Object.getPrototypeOf(proto);
  }
}

【实战】Axios 取消请求 - CancelToken - 简书 (jianshu.com)

// request.js
import axios from 'axios'

const cancelToken = {} // 数组用于储存每个请求的取消函数和请求标识
const CancelToken = axios.CancelToken


// 取消被多次发起的请求，保留最后一个
const cancelAjax = (type, config) => {
    const key = getUrlKey(config)
    if (type === 'check') {
        if (!cancelToken[key]) {
            config.cancelToken = new CancelToken((c) => {
                cancelToken[key] = c
            })
        } else {
            cancelToken[key]()
        }
    } else if (type === 'remove') {
        delete cancelToken[key]
    }
}

// 允许多次重复调用的API白名单
const whiteList = []

// request interceptor
service.interceptors.request.use((config) => {
    if (!whiteList.includes(config.url)) {
        cancelAjax('check', config)
    }
    return config
}, err => {
    console.log('request intercepter err:' + err) // 打印错误
    return Promise.reject(err)
})

// response interceptor
service.interceptors.response.use(
    async response => {
        cancelAjax('remove', response.config)
        const res = response.data
        ...
    },
    err => {
        console.log('response intercepter err: ' + err) // 打印错误
        ...
        return Promise.reject(err)
    }
)

  • 函数柯力化

    function add(a, b, c, d) {
	return a * b * c * d
}

function curry(fn) {
	return function curried(...args) {
		if (args.length >= fn.length) {
			return fn.apply(this, args)
		} else {
			return function (...moreArgs) {
				return curried.apply(this, args.concat(moreArgs))
			}
		}
	}
}

const curriedAdd = curry(add)
const res = curriedAdd(2)(3)(4)(5)
