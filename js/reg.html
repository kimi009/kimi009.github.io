a — 匹配一个字符a(不能匹配 b,  aa等等.)
abc — 匹配 a, 其次 b, 最后  c.
a* — 匹配0个或者多个字符 a (+ 代表至少匹配一个或者多个).
[^a] — 匹配一个字符，但它不能是a.
a|b — 匹配一个字符 a 或者 b.
[abc] — 匹配一个字符，它可以是a,b或c.
[^abc] — 匹配一个字符，但它不可以是a,b或c.
[a-z] — 匹配字符范围 a-z且全部小写  (你可以使用 [A-Za-z] 涵盖大小写, 或 [A-Z] 来限制必须大写).
a.c — 匹配字符 a,中间匹配任意一个字符,最后匹配字符 c.
a{5} — 匹配字符 a五次.
a{5,7} — 匹配字符 a五到七次,不能多或者少.



/**
 * 金额千分位格式化
 * @param {*} amount
 * @param {*} returnIs 是否原样返回
 * @param {*} precision 精度
 * @returns
 */
export const formatAmount = (amount, precision = 2) => {
	if (amount == null || amount == undefined) {
		return null
	}
	amount = amount.toString()

	let [pre, aft] = amount.split('.')
	//处理整数部分
	pre = pre.replace(/(\d)(?=(\d{3})+(?!\d))/g, `$1,`)
	//处理小数部分
	aft = aft ?? ''
	if (precision > 0) {
		//需要精度
		if (aft.length > precision) {
			aft = aft.slice(0, precision)
		} else {
			//补全精度
			aft = aft.padEnd(precision, '0')
		}
		aft = aft.padStart(aft.length + 1, '.')
	} else {
		//不需要精度
		aft = ''
	}
	return `${pre}${aft}`
}
