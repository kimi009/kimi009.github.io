webcomponent 容器 #

什么是 web components？ - 知乎 (zhihu.com)

一、使用方法 #

  • 1.1、创建一个类或函数来指定 web 组件的功能，推荐请使用 ECMAScript 2015 的类语法。
  • 1.2、使用 CustomElementRegistry.define() 方法注册自己的新自定义元素 ，并向其传递要定义的元素名称、指定元素功能的类、以及可选的其所继承自的元素。
  • 1.3、如果需要的话，使用Element.attachShadow() 方法将一个shadow DOM附加到自定义元素上。使用原生的DOM方法向shadow DOM中添加子元素、事件监听器等。
  • 1.4、如果需要的话，使用 <template>和<slot> 定义一个HTML模板。再次使用常规DOM方法克隆模板并将其附加到shadow DOM中。
  • 1.5、最后在页面中使用我们的自定义元素，就像使用原生HTML元素一样

初次使用 #

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>webcomponent</title>
  </head>
  <body>
    <company-card></company-card>
  </body>
  <script>
    class CompanyCard extends HTMLElement {
      constructor() {
        super();
        console.log(this); //this表示当前的实例自定义元素
        //开始创建元素
        let image = document.createElement("img");
        image.src = "./1.png";
        image.style.width = "200px";
        image.style.height = "200px";
        image.classList.add("image");

        let container = document.createElement("div");
        container.classList.add("container");

        let name = document.createElement("p");
        name.classList.add("username");
        name.innerText = "测试webcomponent";

        let email = document.createElement("p");
        email.classList.add("email");
        email.innerText = "kimi@nb.com";

        let btn = document.createElement("button");
        btn.classList.add("button");
        btn.innerText = "开始";

        container.append(image, name, email);

        this.appendChild(container);
      }
    }

    //使用customerElements.define()方法，将页面中自定义的组件和
    //类关联起来
    window.customElements.define("company-card", CompanyCard);
  </script>
</html>

  • :host

    :host CSS 伪类选择包含其内部使用的 CSS 的 shadow DOM 的根元素 - 换句话说，这允许你从其 shadow DOM 中选择一个自定义元素。

    https://developer.mozilla.org/zh-CN/docs/Web/CSS/:host

带模板的 shadow #

<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>webcomponent</title>
  </head>
  <body>
    <p class="name">我是shadow外的元素</p>
    <template id="userCard">
      <style>
        :host {
          display: flex;
          justify-content: flex-start;
          align-items: center;
          flex-direction: column;
          box-shadow: 1px 1px 5px rgba(0, 0, 0, 0.1);
          font-size: 12px;
        }
        .image {
          flex: 0 0 auto;
          border-radius: 5px;
        }
        .container {
          box-sizing: border-box;
          height: 160px;
          padding: 20px;
        }
        .container > .name {
          color: aqua;
        }
        .container > .ipt {
          width: 100px;
        }
        .btn {
          width: 60px;
          height: 25px;
        }
      </style>
      <img class="image" />
      <div class="container">
        <p class="name"></p>
        <input type="text" class="ipt" />
        <button class="btn">提交</button>
      </div>
    </template>
    <user-card image="./1.png" name="大名："></user-card>
  </body>
  <script>
    class UserCard extends HTMLElement {
      constructor() {
        super();
        //自定义元素的this.attachShadow方法，开启Shadow DOM
        //参数表示shadow dom 是封闭的，不允许外部访问
        let shadow = this.attachShadow({ mode: "closed" });
        let tem = document.getElementById("userCard");
        console.log(42, tem, tem.content);
        let content = tem.content.cloneNode(true);

        let imgEle = content.querySelector("img");
        imgEle.setAttribute("src", this.getAttribute("image"));

        imgEle.setAttribute("style", "width:46px;height:20px;");
        content.querySelector(".container>.name").innerText =
          this.getAttribute("name");

        shadow.appendChild(content);

        shadow.querySelector("button").onclick = function () {
          // console.log(document.querySelector('.ipt')) //拿不到，shadow是隔离的
          console.log(this.parentElement.querySelector(".ipt").value); //可以拿到
        };
      }
      connectedCallback() {
        console.log("当 custom element 首次被插入文档dom时，被调用");
      }
      disconnectedCallback() {
        console.log("当 custom element 从文档中删除时，被调用");
      }
      adoptedCallback() {
        console.log("当 custom element 被移到新的文档中时，被调用");
      }
      attributeChangedCallback() {
        console.log("当 custom element 增加，删除，修改自身属性时，被调用");
      }
    }

    //使用customerElements.define()方法，将页面中自定义的组件和
    //类关联起来
    window.customElements.define("user-card", UserCard);
  </script>
</html>
